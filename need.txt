#!/bin/bash
# auto.sh - автоматическая сборка и запуск

set -e

echo "🔧 Сборка утилит..."
mkdir -p bin

# Компиляция всех утилит
for dir in */; do
    dir="${dir%/}"
    if [ -f "$dir/main.go" ]; then
        echo "  Компиляция: $dir"
        go build -o "bin/$dir" "$dir/main.go"
    fi
done

# Добавляем bin в PATH
export PATH="$PWD/bin:$PATH"

echo "✅ Готово! Теперь можно использовать команды."
echo ""
echo "Примеры:"
echo "  ls -la"
echo "  pwd"
echo "  cat test.txt"
echo ""
echo "Чтобы выйти, нажмите Ctrl+C или введите 'exit'"
echo ""

# Запускаем интерактивную сессию
exec bash --norc --noprofile -i
